Script de Limpieza Seguro para Windows

Copia el siguiente código y pégalo en tu consola de PowerShell ejecutada como administrador:

# ==============================================
# Limpieza Segura y Optimizada de Windows
# ==============================================

# Verificar privilegios de administrador
$mr = New-Object Security.Principal.WindowsPrincipal([Security.Principal.WindowsIdentity]::GetCurrent())
if (-not $mr.IsInRole([Security.Principal.WindowsBuiltInRole]::Administrator)) {
    Write-Warning "Este script debe ejecutarse con permisos de Administrador."
    exit 1
}

# Función para limpiar y reportar estado
function Invoke-Clean {
    param(
        [string]$Name,
        [scriptblock]$Action
    )
    Write-Host "Iniciando: $Name"
    try {
        & $Action
        Write-Host "✔ $Name completado."
    } catch {
        Write-Warning "✖ Error en $Name: $_"
    }
}

# Opciones de ejecución
$WhatIfPreference = $false  # Cambiar a $true para simular acciones
$VerbosePreference = 'Continue'

Invoke-Clean "Archivos temporales de usuario" {
    Remove-Item "$env:TEMP\*" -Recurse -Force -ErrorAction Stop
}
Invoke-Clean "Archivos temporales del sistema" {
    Remove-Item "C:\Windows\Temp\*" -Recurse -Force -ErrorAction Stop
}
Invoke-Clean "Papelera de reciclaje" {
    Clear-RecycleBin -Force -ErrorAction Stop
}
Invoke-Clean "Caché de actualizaciones de Windows" {
    Stop-Service wuauserv -Force -ErrorAction Stop
    Remove-Item "C:\Windows\SoftwareDistribution\Download\*" -Recurse -Force -ErrorAction Stop
    Start-Service wuauserv -ErrorAction Stop
}
Invoke-Clean "Caché Prefetch" {
    Remove-Item "C:\Windows\Prefetch\*" -Recurse -Force -ErrorAction Stop
}
Invoke-Clean "Accesos directos recientes" {
    Remove-Item "$env:APPDATA\Microsoft\Windows\Recent\*" -Recurse -Force -ErrorAction Stop
}
Invoke-Clean "Caché de miniaturas" {
    ie4uinit.exe -ClearIconCache
}
Invoke-Clean "Historial de PowerShell" {
    Remove-Item (Get-PSReadlineOption).HistorySavePath -ErrorAction Stop
}
Invoke-Clean "Caché de navegadores (Chrome, Edge)" {
    $localApp = $env:LOCALAPPDATA
    Remove-Item "$localApp\Google\Chrome\User Data\Default\Cache\*" -Recurse -Force -ErrorAction Stop
    Remove-Item "$localApp\Microsoft\Edge\User Data\Default\Cache\*" -Recurse -Force -ErrorAction Stop
}
Invoke-Clean "Logs de eventos" {
    wevtutil el | Foreach-Object { wevtutil cl $_ }
}
Invoke-Clean "Caché DNS" {
    ipconfig /flushdns
}
Invoke-Clean "ComponentStoreCleanup" {
    Dism.exe /Online /Cleanup-Image /StartComponentCleanup /ResetBase
}
Invoke-Clean "Puntos de restauración antiguos" {
    vssadmin delete shadows /for=C: /oldest
}
Invoke-Clean "Caché instalación de controladores" {
    Remove-Item "C:\Windows\System32\DriverStore\Temp\*" -Recurse -Force -ErrorAction Stop
}
Invoke-Clean "Archivos de diagnóstico" {
    Remove-Item "$env:ProgramData\Microsoft\Diagnosis\*" -Recurse -Force -ErrorAction Stop
}
Invoke-Clean "Caché de minivolcado de fallos" {
    Remove-Item "$env:LOCALAPPDATA\CrashDumps\*" -Recurse -Force -ErrorAction Stop
}
Invoke-Clean "OneDrive y sync leftovers" {
    Remove-Item "$env:UserProfile\OneDriveTemp\*" -Recurse -Force -ErrorAction SilentlyContinue
}

Write-Host ""
Write-Host "🎉 Limpieza completa. ¡Tu sistema está listo!"
